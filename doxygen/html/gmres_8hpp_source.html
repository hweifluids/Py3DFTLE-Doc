278 void updateSolution(const int gmresUpdateSize, bool runPreco = true)
279 {
280 for (int k = gmresUpdateSize - 1; k >= 0; --k) {
281 auto y = h_y.template ptr<T>();
282
283 y[k] = s[k];
284
285 for (int m = k + 1; m < gmresUpdateSize; ++m) {
286 y[k] -= H[k + m * (nRestartVectors + 1)] * y[m];
287 }
288
289 y[k] /= (H[k + k * (nRestartVectors + 1)] + this->tiny);
290 }
291
292 o_y.copyFrom(h_y, gmresUpdateSize);
293
294 // xCorr = sum_j y[j] * z[j];
295 correctionKernel(this->Nlocal,
296 this->fieldOffset,
297 gmresUpdateSize,
298 o_y,
299 (flexible) ? o_Z : o_V,
300 (flexible) ? o_xCorr : o_w);
301
302 if (!flexible) {
303 if (preco && runPreco) {
304 preco(o_w, o_xCorr);
305 } else {
306 o_xCorr.copyFrom(o_w);
307 }
308 }
309
310 // x += xCorr
311 updateSolutionKernel(this->Nlocal, this->fieldOffset, o_xCorr, o_x);
312
313 auto flopCount =

334 platform->linAlg->axpbyMany<T>(this->Nlocal,

344 // right preconditioning: z_k = M^{-1} v_k
345 auto o_zk = [&]() {
346 const auto n = o_w.size();

348 if (preco) {
349 auto o_zk = flexible ? o_Z.slice(i * offset, n) : o_Z.slice(0, n);
350 preco(o_vk, o_zk);

352 } else {

354 }
355 }();

359 o_weight,
360 o_zk,
361 platform->comm.mpiComm());

363 platform->linAlg->add(o_zk.size(), -dotp / weightSum, o_zk);
364 }
365
366 Ax(o_zk, o_w);
367
368 // 1 pass classical Gram-Schmidt (project new solution vector o_w onto o_V)
369 {
370#if USE_WEIGHTED_INNER_PROD_MULTI_DEVICE
371 platform->linAlg->weightedInnerProdMulti<T>(this->Nlocal,
372 (i + 1),
373 this->Nfields,
374 this->fieldOffset,
375 o_weight,
376 o_V,
377 o_w,
378 platform->comm.mpiComm(),

380 o_y.copyTo(h_y, (i + 1));
381#else

383 (i + 1),
384 this->Nfields,
385 this->fieldOffset,
386 o_weight,
387 o_V,
388 o_w,
389 platform->comm.mpiComm(),

392#endif

394 // orthogonalize o_w against previous o_V
395 gsOrthoKernel(Nblock, this->Nlocal, this->fieldOffset, (i + 1), o_weight, o_y, o_V, o_w, o_scratch);
396
397 double flopCount = FPfactor * 5 * (i + 1) * this->Nfields * static_cast<double>(this->Nlocal);
398 platform->flopCounter->add("gramSchmidt", flopCount);
399 }
400
401 // normalize
402 auto nw = [&]() {
403 dfloat norm = 0;
404 if (platform->serial()) {
405 norm = o_scratch.ptr<T>()[0];
406 } else {
407 o_scratch.copyTo(h_scratch);
408 auto scratch = h_scratch.template ptr<T>();
409 for (int k = 0; k < h_scratch.size(); ++k) {
410 norm += scratch[k];
411 }
412 }
413 MPI_Allreduce(MPI_IN_PLACE, &norm, 1, MPI_DFLOAT, MPI_SUM, platform->comm.mpiComm());
414 return std::sqrt(norm);
415 }();
416
417 if (i < nRestartVectors - 1) {
418 auto o_Vi = o_V.slice((i + 1) * offset);
419 platform->linAlg->axpbyMany<T>(this->Nlocal,
420 this->Nfields,
421 this->fieldOffset,
422 1. / (nw + this->tiny),
423 o_w,
424 0,
425 o_Vi);
426 }
427
428 // apply Givens rotation
429 H[i + 1 + i * (nRestartVectors + 1)] = nw;
430 for (int k = 0; k <= i; ++k) {
431 H[k + i * (nRestartVectors + 1)] = h_y.template ptr<T>()[k];
432 }
433
434 for (int k = 0; k < i; ++k) {
435 const dfloat h1 = H[k + i * (nRestartVectors + 1)];
436 const dfloat h2 = H[k + 1 + i * (nRestartVectors + 1)];
437
438 H[k + i * (nRestartVectors + 1)] = cs[k] * h1 + sn[k] * h2;
439 H[k + 1 + i * (nRestartVectors + 1)] = -sn[k] * h1 + cs[k] * h2;
440 }
441
442 // form i-th rotation matrix
443 const auto h1 = H[i + i * (nRestartVectors + 1)];
444 const auto h2 = H[i + 1 + i * (nRestartVectors + 1)];
445#if 1
446 const auto r = std::hypot(h1, h2);
447 cs[i] = (r == 0) ? 1 : h1 / r;
448 sn[i] = (r == 0) ? 0 : h2 / r;
449 H[i + i * (nRestartVectors + 1)] = r;
450 H[i + 1 + i * (nRestartVectors + 1)] = 0;
451#else
452 const auto hr = 1 / (std::sqrt(h1 * h1 + h2 * h2) + this->tiny);
453 cs[i] = h1 * hr;
454 sn[i] = h2 * hr;
455
456 H[i + i * (nRestartVectors + 1)] = cs[i] * h1 + sn[i] * h2;
457 H[i + 1 + i * (nRestartVectors + 1)] = 0;
458#endif
459 // approximate residual norm
460 s[i + 1] = -sn[i] * s[i];
461 s[i] = cs[i] * s[i];
462
463 if (!iR) {
464 this->rNorm = std::abs(s[i + 1]);
465
466 if (platform->comm.mpiRank() == 0) {
467 nekrsCheck(!std::isfinite(this->rNorm),
468 MPI_COMM_SELF,
469 EXIT_FAILURE,
470 "%s invalid resiual norm while running linear solver!\n",
471 this->_name.c_str());
472 }
473
474 const auto iter = iter0 + i + 1;
475 if (platform->verbose() && platform->comm.mpiRank() == 0) {
476 printf("it %d r norm %.15e\n", iter, this->rNorm);
477 }
478
479 if (this->rNorm < tol || iter == maxIter) {
480 return i + 1;
481 }
482 }
483 }
484
485 return nRestartVectors;
