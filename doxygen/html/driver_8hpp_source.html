77cmdOptions* processCmdLineOptions(int argc, char** argv, const MPI_Comm &comm)
78{
79 int rank, size;
80 MPI_Comm_rank(comm, &rank);
81 MPI_Comm_size(comm, &size);
82
83 cmdOptions* cmdOpt = new cmdOptions();
84
85 int err = 0;
86 int printHelp = 0;
87 std::string helpCat;
88
89 if (rank == 0) {
90 while(1) {
91 static struct option long_options[] =
92 {
93 {"setup", required_argument, 0, 's'},
94 {"cimode", required_argument, 0, 'c'},
95 {"build-only", optional_argument, 0, 'b'},
96 {"debug", no_argument, 0, 'd'},
97 {"attach", no_argument, 0, 'a'},
98 {"backend", required_argument, 0, 't'},
99 {"device-id", required_argument, 0, 'i'},
100 {"output", required_argument, 0, 'r'},
101 {"help", optional_argument, 0, 'h'},
102 {0, 0, 0, 0}
103 };
104 int option_index = 0;
105 int c = getopt_long (argc, argv, "", long_options, &option_index);
106
107 if (c == -1)
108 break;
109
110 switch(c) {
111 case 's':
112 cmdOpt->setupFile.assign(optarg);
113 if (cmdOpt->setupFile.find(".par") != std::string::npos)
114 cmdOpt->setupFile.erase(cmdOpt->setupFile.find(".par"), std::string::npos);
115 if (cmdOpt->setupFile.substr(cmdOpt->setupFile.find_last_of(".") + 1) == "sess") {
116 cmdOpt->multiSessionFile = cmdOpt->setupFile;
117 cmdOpt->setupFile.clear();
118 }
119 break;
120 case 'b':
121 cmdOpt->buildOnly = 1;
122 cmdOpt->sizeTarget = size;
123 if(!optarg && argv[optind] != NULL && argv[optind][0] != '-')
124 cmdOpt->sizeTarget = std::stoi(argv[optind++]);
125 break;
126 case 'c':
127 cmdOpt->ciMode = atoi(optarg);
128 if (cmdOpt->ciMode < 0) {
129 std::cout << "ERROR: ci test id has to be >= 0!\n";
130 printHelp = 1;
131 }
132 break;
133 case 'd':
134 cmdOpt->debug = 1;
135 break;
136 case 'a':
137 cmdOpt->attach = 1;
138 cmdOpt->debug = 0;
139 break;
140 case 'i':
141 cmdOpt->deviceID.assign(optarg);
142 break;
143 case 't':
144 cmdOpt->backend.assign(optarg);
145 break;
146 case 'r':
147 cmdOpt->redirectOutput.assign(optarg);
148 break;
149 case 'h':
150 printHelp++;
151 if(!optarg && argv[optind] != NULL && argv[optind][0] != '-')
152 helpCat.assign(argv[optind++]);
153 break;
154 default:
155 err = 1;
156 }
157 }
158 }
159
160 for(auto opt: {&cmdOpt->multiSessionFile, &cmdOpt->setupFile, &cmdOpt->deviceID, &cmdOpt->backend, &cmdOpt->redirectOutput})
161 {
162 int bufSize = opt->size() + 1;
163 MPI_Bcast(&bufSize, 1, MPI_INT, 0, comm);
164 auto buf = (char*) std::calloc(bufSize, sizeof(char));
165 if(rank == 0) std::strcpy(buf, opt->c_str());
166 MPI_Bcast(buf, bufSize, MPI_BYTE, 0, comm);
167 opt->assign(buf);
168 free(buf);
169 }
170
171 MPI_Bcast(&cmdOpt->buildOnly, sizeof(cmdOpt->buildOnly), MPI_BYTE, 0, comm);
172 MPI_Bcast(&cmdOpt->sizeTarget, sizeof(cmdOpt->sizeTarget), MPI_BYTE, 0, comm);
173 MPI_Bcast(&cmdOpt->ciMode, sizeof(cmdOpt->ciMode), MPI_BYTE, 0, comm);
174 MPI_Bcast(&cmdOpt->debug, sizeof(cmdOpt->debug), MPI_BYTE, 0, comm);
175 MPI_Bcast(&cmdOpt->attach, sizeof(cmdOpt->attach), MPI_BYTE, 0, comm);
176
177 if (cmdOpt->debug) {
178 std::feclearexcept(FE_ALL_EXCEPT);
179 std::fegetexceptflag(&flag, FE_ALL_EXCEPT);
180 }
181
182 if(cmdOpt->setupFile.empty() && cmdOpt->multiSessionFile.empty())
183 printHelp++;
184
185 MPI_Bcast(&printHelp, sizeof(printHelp), MPI_BYTE, 0, comm);
186 MPI_Bcast(&err, sizeof(err), MPI_BYTE, 0, comm);
187 if (err | printHelp) {
188 if (rank == 0) {
189 auto print = [&](const std::string& txtFile)
190 {
191 std::string installDir;
192 installDir.assign(getenv("NEKRS_HOME"));
193 std::ifstream f(installDir + "/doc/" + txtFile);
194 if (f.is_open()) std::cout << f.rdbuf();
195 f.close();
196 };
197
198 if (helpCat == "par") {
199 print("parHelp.txt");
200 } else if (helpCat == "env") {
201 print("envHelp.txt");
202 } else {
203 std::cout << "usage: ./nekrs "
204 << "[ --help <par|env> ] "
205 << "--setup <par|sess file> "
206 << "[ --build-only <#procs> ] [ --cimode <id> ] [ --debug ] [ --attach ] "
207 << "[ --backend <CPU|CUDA|HIP|DPCPP|OPENCL> ] [ --device-id <id|LOCAL-RANK> ] "
208 << "[ --output <local-path-to-logfile>]"
209 << "\n";
210 }
211 }
212 MPI_Finalize();
213 exit((err) ? EXIT_FAILURE : EXIT_SUCCESS);
214 }
215
216 return cmdOpt;
217}

219std::map<std::string, std::map<std::string, std::string>> readPar(const std::string &_setupFile, MPI_Comm comm)
220{
221 int rank;
222 MPI_Comm_rank(comm, &rank);
223
224 const auto setupFile = _setupFile + ".par";
225
226 if (rank == 0)
227 std::cout << "reading " << setupFile << std::endl;
228
229 int err = 0;
230 if (rank == 0) {
231 if (!file_exist(setupFile)) {
232 std::cerr << "Cannot find setup file " << setupFile << std::endl;
233 err++;
234 }
235 }
236 MPI_Allreduce(MPI_IN_PLACE, &err, 1, MPI_INT, MPI_MAX, comm);
237 if (err) {
238 MPI_Abort(comm, EXIT_FAILURE);
239 }

242 if (rank == 0) {
243 fsize = file_size(setupFile);

245 MPI_Bcast(&fsize, sizeof(fsize), MPI_BYTE, 0, comm);

247 if (rank == 0) {
248 std::ifstream input(setupFile, std::ifstream::binary);
249 input.read(fileBuf, fsize);
250 input.close();
251 }
252 MPI_Bcast(fileBuf, fsize, MPI_CHAR, 0, comm);
253
254 auto par = new inipp::Ini();
255
256 std::stringstream is;
257 is.write(fileBuf, fsize);
258
259 par->parse(is);
260 par->interpolate();
261

265MPI_Comm setupSession(cmdOptions *cmdOpt, const MPI_Comm &comm)
266{
267 int rank, size;
268 MPI_Comm_rank(comm, &rank);
269 MPI_Comm_size(comm, &size);
270 MPI_Comm newComm = comm;
271
272 if(cmdOpt->multiSessionFile.size()) {
273 std::string multiSessionFileContent;
274
275 if(rank == 0) {
276 std::ifstream f(cmdOpt->multiSessionFile);
277 if (!f) {
278 std::cout << "FATAL ERROR: Cannot find sess file "
279 << cmdOpt->multiSessionFile << "!\n";
280 fflush(stdout);
281 MPI_Abort(comm, EXIT_FAILURE);
282 }
283 std::ostringstream ss;
284 if (f.is_open()) ss << f.rdbuf();
285 f.close();
286 multiSessionFileContent = ss.str();
287 }
288 int bufSize = multiSessionFileContent.size() + 1;
289 MPI_Bcast(&bufSize, sizeof(bufSize), MPI_BYTE, 0, comm);
290 char* buf = (char*) malloc(bufSize * sizeof(char));
291 strcpy(buf, multiSessionFileContent.c_str());
292 MPI_Bcast(buf, bufSize * sizeof(char), MPI_BYTE, 0, comm);
293 multiSessionFileContent = std::string(buf);
294 free(buf);
295
296 auto serializeString = [](const std::string sin, char dlim)
297 {
298 std::vector<std::string> slist;
299 std::string s(sin);
300 s.erase(std::remove_if(s.begin(), s.end(), ::isspace), s.end());
301 std::stringstream ss;
302 ss.str(s);
303 while (ss.good()) {
304 std::string substr;
305 std::getline(ss, substr, dlim);
306 if(!substr.empty()) slist.push_back(substr);
307 }
308 return slist;
309 };
310
311 auto list = serializeString(multiSessionFileContent, ';');
312 auto sessionList = new session_t[list.size()];
313
314 int nSessions = 0;
315 int rankSum = 0;
316 for(std::string s : list) {
317 auto items = serializeString(s,':');
318 if(items.size() != 2) {
319 if(rank == 0) std::cout << "FATAL ERROR: invalid sess file entry!\n";
320 fflush(stdout);
321 MPI_Abort(comm, EXIT_FAILURE);
322 }
323 sessionList[nSessions].setupFile = items[0];
324 sessionList[nSessions].size = std::stoi(items[1]);
325 rankSum += sessionList[nSessions].size;
326 nSessions++;
327 }
328
329 int err = 0;
330 if(rankSum != size) err = 1;
331 MPI_Allreduce(MPI_IN_PLACE, &err, 1, MPI_INT, MPI_SUM, comm);
332 if(err) {
333 if(rank == 0) std::cout << "FATAL ERROR: size of sub-communicators does not match parent!\n";
334 fflush(stdout);
335 MPI_Abort(comm, EXIT_FAILURE);
336 }
337
338 int color = MPI_UNDEFINED;
339 int rankOffsetSession = 0;
340 for(int i = 0; i < nSessions; i++) {
341 if(rank - rankOffsetSession < sessionList[i].size) {
342 color = i;
343 break;
344 }
345 rankOffsetSession += sessionList[i].size;
346 }
347
348 int rankGlobal, sizeGlobal;
349 MPI_Comm_rank(comm, &rankGlobal);
350 MPI_Comm_size(comm, &sizeGlobal);
351
352 MPI_Comm_split(comm, color, rankGlobal, &newComm);
353
354 MPI_Comm_rank(newComm, &rank);
355 MPI_Comm_size(newComm, &size);
356
357 cmdOpt->setupFile = sessionList[color].setupFile;
358 cmdOpt->sizeTarget = size;
359 cmdOpt->sessionID = color;
360 cmdOpt->nSessions = nSessions;
361
362 delete[] sessionList;
363
364 if(cmdOpt->debug) {
365 std::cout << "globalRank:" << rankGlobal << " localRank: " << rank << " pid: " << ::getpid()
366 << " commSize: " << size << " setupFile:" << cmdOpt->setupFile << "\n";
367 }
368 fflush(stdout);
369 MPI_Barrier(comm);
370
371 const std::string outputFile =
372 cmdOpt->setupFile.substr(0, cmdOpt->setupFile.rfind('/', cmdOpt->setupFile.length())) + "/logfile";
373 if(rank == 0) {
374 std::cout << "redirecting session" << cmdOpt->sessionID << " output to " << outputFile << " ...\n";
375 }
376 const int fd = open(outputFile.c_str(), O_WRONLY|O_CREAT|O_APPEND, S_IWUSR|S_IRUSR);
377 dup2(fd, fileno(stderr));
378 dup2(fd, fileno(stdout));
379 }
380 return newComm;
381}
