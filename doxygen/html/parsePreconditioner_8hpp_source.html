184 append_error("Overlaping coarse grid solve is only supported with boomerAMG!\n");
185 }

204void parseSmoother(const int rank, setupAide &options, inipp::Ini *ini, std::string parScope)
205{
206 std::string p_smoother;
207
208 std::string parSection = upperCase(parPrefixFromParSection(parScope));
209
210 if (options.compareArgs(parSection + "PRECONDITIONER", "MULTIGRID")) {
211 options.setArgs(parSection + "MULTIGRID SMOOTHER", "FOURTHOPTCHEBYSHEV+DAMPEDJACOBI");
212 options.setArgs(parSection + "MULTIGRID CHEBYSHEV DEGREE", "1");
213 options.setArgs(parSection + "MULTIGRID CHEBYSHEV MAX EIGENVALUE BOUND FACTOR", "1.1");
214 if (parScope == "fluid pressure") {
215 if (options.compareArgs(parSection + "PRECONDITIONER", "MULTIGRID+SEMFEM")) {
216 options.setArgs(parSection + "MULTIGRID CHEBYSHEV DEGREE", "2");
217 } else {
218 options.setArgs(parSection + "MULTIGRID SMOOTHER", "FOURTHOPTCHEBYSHEV+ASM");
219 options.setArgs(parSection + "MULTIGRID CHEBYSHEV DEGREE", "3");
220 }
221 }
222 }
223
224 if (!ini->extract(parScope, "smoothertype", p_smoother)) {
225 return;
226 }
227
228 std::string p_preconditioner;
229 ini->extract(parScope, "preconditioner", p_preconditioner);
230
231 const std::vector<std::string> validValues = {
232 {"asm"},
233 {"ras"},
234 {"cheby"},
235 {"fourthcheby"},
236 {"fourthoptcheby"},
237 {"jac"},
238 {"degree"},
239 {"mineigenvalueboundfactor"},
240 {"maxeigenvalueboundfactor"},
241 };
242
243 {
244 const std::vector<std::string> list = serializeString(p_smoother, '+');
245 for (const std::string s : list) {
246 checkValidity(rank, validValues, s);
247 }
248 }
249
250 if (options.compareArgs(parSection + "PRECONDITIONER", "MULTIGRID")) {
251 std::vector<std::string> list;
252 list = serializeString(p_smoother, '+');
253
254 if (p_smoother.find("cheb") != std::string::npos) {
255 bool surrogateSmootherSet = false;
256 std::string chebyshevType = "";
257 if (p_smoother.find("fourthopt") != std::string::npos) {
258 chebyshevType = "FOURTHOPTCHEBYSHEV";
259 } else if (p_smoother.find("fourth") != std::string::npos) {
260 chebyshevType = "FOURTHCHEBYSHEV";
261 } else {
262 // using 1st-kind Chebyshev, so set a reasonable lmin multiplier
263 chebyshevType = "CHEBYSHEV";
264 options.setArgs(parSection + "MULTIGRID CHEBYSHEV MIN EIGENVALUE BOUND FACTOR", "0.1");
265 }
266
267 for (std::string s : list) {
268 const auto degreeStr = parseValueForKey(s, "degree");
269 if (!degreeStr.empty()) {
270 options.setArgs(parSection + "MULTIGRID CHEBYSHEV DEGREE", degreeStr);
271 }
272
273 const auto minEigBoundStr = parseValueForKey(s, "mineigenvalueboundfactor");
274 if (!minEigBoundStr.empty()) {
275 if (chebyshevType.find("FOURTH") != std::string::npos) {
276 append_error(
277 "minEigenvalueBoundFactor not supported for 4th kind or Opt. 4th kind Chebyshev smoother!\n");
278 }
279 options.setArgs(parSection + "MULTIGRID CHEBYSHEV MIN EIGENVALUE BOUND FACTOR", minEigBoundStr);
280 }
281
282 const auto maxEigBoundStr = parseValueForKey(s, "maxeigenvalueboundfactor");
283 if (!maxEigBoundStr.empty()) {
284 options.setArgs(parSection + "MULTIGRID CHEBYSHEV MAX EIGENVALUE BOUND FACTOR", maxEigBoundStr);
285 }
286
287 if (s.find("jac") != std::string::npos) {
288 surrogateSmootherSet = true;
289 options.setArgs(parSection + "MULTIGRID SMOOTHER", chebyshevType + "+DAMPEDJACOBI");
290 } else if (s.find("asm") != std::string::npos) {
291 surrogateSmootherSet = true;
292 options.setArgs(parSection + "MULTIGRID SMOOTHER", chebyshevType + "+ASM");
293 } else if (s.find("ras") != std::string::npos) {
294 surrogateSmootherSet = true;
295 options.setArgs(parSection + "MULTIGRID SMOOTHER", chebyshevType + "+RAS");
296 }
297 }
298
299 if (!surrogateSmootherSet) {
300 append_error("Inner Chebyshev smoother not set");
301 }
302 return;
303 }
304
305 // Non-Chebyshev smoothers
306 options.removeArgs(parSection + "MULTIGRID CHEBYSHEV DEGREE");
307 options.removeArgs(parSection + "MULTIGRID CHEBYSHEV MAX EIGENVALUE BOUND FACTOR");
308 if (p_smoother.find("asm") == 0) {
309 options.setArgs(parSection + "MULTIGRID SMOOTHER", "ASM");
310 } else if (p_smoother.find("ras") == 0) {
311 options.setArgs(parSection + "MULTIGRID SMOOTHER", "RAS");
312 } else if (p_smoother.find("jac") == 0) {
313 append_error("Jacobi smoother requires Chebyshev");
314 options.setArgs(parSection + "MULTIGRID SMOOTHER", "DAMPEDJACOBI");
315 } else {
316 append_error("Unknown ::smootherType");
317 }
318 }

321void parsePreconditioner(const int rank, setupAide &options, inipp::Ini *ini, std::string parScope)
322{
323 const std::vector<std::string> validValues = {
324 {"none"},
325 {"jac"},
326 {"semfem"},
327 {"femsem"},
328 {"pmg"},
329 {"multigrid"},
330 {"additive"},
331 {"multiplicative"},
332 };
333
334 std::string parSection = upperCase(parPrefixFromParSection(parScope));
335
336 std::string p_preconditioner;
337 if (!ini->extract(parScope, "preconditioner", p_preconditioner)) {
338 p_preconditioner = "jac";
339 if (parScope.find("fluid pressure") != std::string::npos) {
340 p_preconditioner = "multigrid";
341 }
342 }
343
344 const std::vector<std::string> list = serializeString(p_preconditioner, '+');
345 for (std::string s : list) {
346 checkValidity(rank, validValues, s);
347 }
348
349 const auto mg = p_preconditioner.find("pmg") != std::string::npos ||
350 p_preconditioner.find("multigrid") != std::string::npos;
351
352 const auto semfem = p_preconditioner.find("semfem") != std::string::npos ||
353 p_preconditioner.find("femsem") != std::string::npos;
354
355 if (p_preconditioner.find("none") != std::string::npos) {
356 options.setArgs(parSection + "PRECONDITIONER", "NONE");
357 return;
358 } else if (p_preconditioner.find("jac") != std::string::npos) {
359 options.setArgs(parSection + "PRECONDITIONER", "JACOBI");
360 options.setArgs(parSection + "ELLIPTIC PRECO COEFF FIELD", "TRUE");
361 } else if (mg) {
362 options.setArgs(parSection + "PRECONDITIONER", "MULTIGRID");
363 options.setArgs(parSection + "ELLIPTIC PRECO COEFF FIELD", "FALSE");
364 std::string key = "VCYCLE+MULTIPLICATIVE";
365 if (p_preconditioner.find("additive") != std::string::npos) {
366 key = "VCYCLE+ADDITIVE";
367 } else if (p_preconditioner.find("multiplicative") != std::string::npos) {
368 key = "VCYCLE+MULTIPLICATIVE";
369 }
370 options.setArgs(parSection + "MGSOLVER CYCLE", key);
371
372 if (semfem) {
373 options.setArgs(parSection + "PRECONDITIONER", "MULTIGRID+SEMFEM");
374 options.setArgs(parSection + "MGSOLVER CYCLE", "VCYCLE+MULTIPLICATIVE");
375 options.setArgs(parSection + "ELLIPTIC PRECO COEFF FIELD", "FALSE");
376 }
377 } else if (p_preconditioner.find("semfem") != std::string::npos ||
378 p_preconditioner.find("femsem") != std::string::npos) {
379 options.setArgs(parSection + "PRECONDITIONER", "SEMFEM");
380#if 0
381 std::string p_coarseGridDiscretization;
382 if (ini->extract(parScope, "coarsegriddiscretization", p_coarseGridDiscretization)) {
383 if (p_coarseGridDiscretization.find("semfem") != std::string::npos) {
384 smoothed = false;
385 }
386 }
387#endif
388 }
389
390 parseSmoother(rank, options, ini, parScope);
391
392 parseCoarseGridDiscretization(rank, options, ini, parScope);
393
394 if (options.compareArgs(parSection + "PRECONDITIONER", "MULTIGRID") ||
395 options.compareArgs(parSection + "PRECONDITIONER", "SEMFEM")) {
396 parseCoarseSolver(rank, options, ini, parScope);
397 }
398
399 if (options.compareArgs(parSection + "PRECONDITIONER", "MULTIGRID")) {
400 std::string p_mgschedule;
401 if (ini->extract(parScope, "pmgschedule", p_mgschedule)) {
402 const auto semfem = options.getArgs(parSection + "PRECONDITIONER") == "SEMFEM";
403 if (semfem) {
404 append_error("pMGSchedule not supported for preconditioner = semfem.\n");
405 }
406
407 options.setArgs(parSection + "MULTIGRID SCHEDULE", p_mgschedule);
408
409 options.removeArgs(parSection + "MULTIGRID CHEBYSHEV DEGREE");
410
411 // validate multigrid schedule
412 // note: default order here is not actually required
413 auto [scheduleMap, errorString] = ellipticParseMultigridSchedule(p_mgschedule, options, 3);
414 if (!errorString.empty()) {
415 append_error(errorString);
416 }
417
418 int minDegree = std::numeric_limits<int>::max();
419 for (auto &&[cyclePosition, smootherOrder] : scheduleMap) {
420 auto [polyOrder, isDownLeg] = cyclePosition;
421 minDegree = std::min(minDegree, polyOrder);
422 }
423
424 const auto INVALID = std::numeric_limits<int>::lowest();
425
426 // bail if degree is set _and_ it conflicts
427 std::string p_smoother;
428 if (ini->extract(parScope, "smoothertype", p_smoother)) {
429 for (auto &&s : serializeString(p_smoother, '+')) {
430 if (s.find("degree") != std::string::npos) {
431 const auto degreeStr = parseValueForKey(s, "degree");
432 if (!degreeStr.empty()) {
433 const auto specifiedDegree = std::stoi(degreeStr);
434 for (auto &&[cyclePosition, smootherOrder] : scheduleMap) {
435 auto [polyOrder, isDownLeg] = cyclePosition;
436 const bool degreeConflicts = smootherOrder != specifiedDegree;
437 const bool isMinOrder = polyOrder == minDegree;
438 const bool minOrderInvalid = smootherOrder == INVALID;
439
440 if (isMinOrder && minOrderInvalid) {
441 continue;
442 }
443
444 if (degreeConflicts) {
445 append_error(
446 "order specified in pMGSchedule conflicts with that specified in smootherType!\n");
447 }
448 }
449 }
450 }
451 }
452 }
453
454 // bail if coarse degree is set, but we're not smoothing on the coarsest level
455 if (scheduleMap[{minDegree, true}] > 0) {
456
457 if (!options.compareArgs(parSection + "MULTIGRID COARSE SOLVER", "SMOOTHER")) {
458 append_error("specified coarse Chebyshev degree, but coarseSolver=smoother is not set.\n");
459 }
460 }
461 }
462 }
