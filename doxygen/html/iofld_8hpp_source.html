127 hRefineScheduleApply.resize(ndiff, 0);

135 void writeAttribute(const std::string &key_, const std::string &val)
136 {
137 nekrsCheck(!initialized, MPI_COMM_SELF, EXIT_FAILURE, "%s\n", "illegal to call prior to iofld::open()!");
138
139 std::string key = lowerCase(key_);
140
141 if (key == "polynomialorder") {
142 N = std::stoi(val);
143 nekrsCheck(N < 1 || N > 15, MPI_COMM_SELF, EXIT_FAILURE, "invalid polynomial order %d\n", N);
144 } else if (key == "precision") {

153 } else if (key == "outputmesh") {

160 hRefineScheduleSim.push_back(std::stoi(s));

381 template <typename T = std::vector<occa::memory>>
382 std::optional<std::reference_wrapper<T>> inquireVariable(const std::string &name)
383 {
384 nekrsCheck(!initialized, MPI_COMM_SELF, EXIT_FAILURE, "%s\n", "illegal to call prior to iofld::open()!");
385
386 if (userFields.count(name) == 0 && userSingleValues.count(name) == 0) {
387 return std::nullopt;
388 }
389
390 if constexpr (std::is_same_v<T, std::vector<occa::memory>>) {
391 return userFields.at(name);
392 } else {
393 auto visitor = [](const auto &value) -> variantType {
394 if constexpr (std::is_same_v<std::decay_t<decltype(value)>, std::reference_wrapper<int>> ||
395 std::is_same_v<std::decay_t<decltype(value)>, std::reference_wrapper<long long int>> ||
396 std::is_same_v<std::decay_t<decltype(value)>, std::reference_wrapper<float>> ||
397 std::is_same_v<std::decay_t<decltype(value)>, std::reference_wrapper<double>>) {
398 return value;
399 } else {
400 throw std::bad_variant_access(); // Mismatched type
401 }
402 };
403
404 // retrieve the reference from the variant, unwrap it
405 return std::get<std::reference_wrapper<T>>(std::visit(visitor, userSingleValues.at(name))).get();
406 }
